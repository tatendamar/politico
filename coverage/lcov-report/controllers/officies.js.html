let getOffice = (req, res) => {
  const id = req.params.officeId;
 
  let found = office['data'].find(office => {
    return office.id === parseInt(id);
  });
 
  if (found) {
    return res.send({
      status: office.status,
      data: found
    });
  } else {
    return res.send({
      status: 404,
      message: 'Not Found'
    });
  }
};
